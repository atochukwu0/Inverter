39/*
40 * ---------------------------------------------------------------------------------------------------------------------
41 * Helper function to link a converter struct pointer to a CAN message array (uint_8t[8])
42 * This converter struct acts like a "getSigVal()" function.
43 * Usage example:
44 * DBC_linkMsgConverter(DBC_IMED_HELMBUS_MSG_BTN_STATUS, RxMessage);
45 *
46 * -> DBC_IMED_HELMBUS_MSG_BTN_STATUS is a generated CANdb message struct.
47 * -> RxMessage is an uint8_t array[8] as received or sent by the can peripheral.
48 *
49 * Now, the message data can be accessed like this:
50 * uint32_t uid = DBC_IMED_HELMBUS_MSG_BTN_STATUS.getSigVal->Board_UID;
51 *
52 * -> or whatever signals are defined in that message.
53 * ---------------------------------------------------------------------------------------------------------------------
54 */
55#ifndef DBC_linkMsgConverter
56#define DBC_linkMsgConverter(dbc, arr) (dbc).getSigVal = (typeof((dbc).getSigVal)) &(arr)
57#define DBC_setup(dbc, MSG) \
58do { \
59 (MSG).DLC = (dbc).DLC; \
60 (MSG).IDE = (dbc).IDE; \
61 (MSG).RTR = 0; \
62 (MSG).ExtId = (MSG).IDE > 0 ? (dbc).ID : 0; \
63 (MSG).StdId = (MSG).IDE > 0 ? 0 : (dbc).ID; \
64} while (0)
